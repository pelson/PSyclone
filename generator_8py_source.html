66 def handle_script(script_name, psy):
67  '''Loads and applies the specified script to the given psy layer.
68  The 'trans' function of the script is called with psy as parameter.
69  :param script_name: Name of the script to load.
70  :type script_name: string
71  :param psy: The psy layer to which the script is applied.
72  :type psy: :py:class:`psyclone.psyGen.PSy`
73  :raises IOError: If the file is not found.
74  :raises GenerationError: if the file does not have .py extension
75  or can not be imported.
76  :raises GenerationError: if trans() can not be called.
77  :raises GenerationError: if any exception is raised when trans()
78  was called.
79  '''
80  sys_path_appended = False
81  try:
82  # a script has been provided
83  filepath, filename = os.path.split(script_name)
84  if filepath:
85  # a path to a file has been provided
86  # we need to check the file exists
87  if not os.path.isfile(script_name):
88  raise IOError("script file '{0}' not found".
89  format(script_name))
90  # it exists so we need to add the path to the python
91  # search path
92  sys_path_appended = True
93  sys.path.append(filepath)
94  filename, fileext = os.path.splitext(filename)
95  if fileext != '.py':
96  raise GenerationError(
97  "generator: expected the script file '{0}' to have "
98  "the '.py' extension".format(filename))
99  try:
100  transmod = __import__(filename)
101  except ImportError:
102  raise GenerationError(
103  "generator: attempted to import '{0}' but script file "
104  "'{1}' has not been found".
105  format(filename, script_name))
106  except SyntaxError:
107  raise GenerationError(
108  "generator: attempted to import '{0}' but script file "
109  "'{1}' is not valid python".
110  format(filename, script_name))
111  if callable(getattr(transmod, 'trans', None)):
112  try:
113  psy = transmod.trans(psy)
114  except Exception:
115  exc_type, exc_value, exc_traceback = sys.exc_info()
116  lines = traceback.format_exception(exc_type, exc_value,
117  exc_traceback)
118  e_str = '{\n' +\
119  ''.join(' ' + line for line in lines[2:]) + '}'
120  raise GenerationError(
121  "Generator: script file '{0}'\nraised the "
122  "following exception during execution "
123  "...\n{1}\nPlease check your script".format(
124  script_name, e_str))
125  else:
126  raise GenerationError(
127  "generator: attempted to import '{0}' but script file "
128  "'{1}' does not contain a 'trans()' function".
129  format(filename, script_name))
130  except Exception as msg:
131  if sys_path_appended:
132  os.sys.path.pop()
133  raise msg
134  if sys_path_appended:
135  os.sys.path.pop()

138 def generate(filename, api="", kernel_path="", script_name=None,
139  line_length=False,
140  distributed_memory=None,
141  kern_out_path="",
142  kern_naming="multiple"):
143  # pylint: disable=too-many-arguments
144  '''Takes a GungHo algorithm specification as input and outputs the
145  associated generated algorithm and psy codes suitable for
146  compiling with the specified kernel(s) and GungHo
147  infrastructure. Uses the :func:`parse.parse` function to parse the
148  algorithm specification, the :class:`psyGen.PSy` class to generate
149  the PSy code and the :class:`algGen.Alg` class to generate the
150  modified algorithm code.
151 
152  :param str filename: The file containing the algorithm specification.
153  :param str kernel_path: The directory from which to recursively \
154  search for the files containing the kernel \
155  source (if different from the location of the \
156  algorithm specification).
157  :param str script_name: A script file that can apply optimisations \
158  to the PSy layer (can be a path to a file or \
159  a filename that relies on the PYTHONPATH to \
160  find the module).
161  :param bool line_length: A logical flag specifying whether we care \
162  about line lengths being longer than 132 \
163  characters. If so, the input (algorithm \
164  and kernel) code is checked to make sure \
165  that it conforms. The default is False.
166  :param bool distributed_memory: A logical flag specifying whether to \
167  generate distributed memory code. The \
168  default is set in the config.py file.
169  :param str kern_out_path: Directory to which to write transformed \
170  kernel code.
171  :param bool kern_naming: the scheme to use when re-naming transformed \
172  kernels.
173  :return: 2-tuple containing fparser1 ASTs for the algorithm code and \
174  the psy code.
175  :rtype: (:py:class:`fparser.one.block_statements.BeginSource`, \
176  :py:class:`fparser.one.block_statements.Module`)
177 
178  :raises IOError: if the filename or search path do not exist
179  :raises GenerationError: if an invalid API is specified.
180  :raises GenerationError: if an invalid kernel-renaming scheme is specified.
181 
182  For example:
183 
184  >>> from psyclone.generator import generate
185  >>> alg, psy = generate("algspec.f90")
186  >>> alg, psy = generate("algspec.f90", kernel_path="src/kernels")
187  >>> alg, psy = generate("algspec.f90", script_name="optimise.py")
188  >>> alg, psy = generate("algspec.f90", line_length=True)
189  >>> alg, psy = generate("algspec.f90", distributed_memory=False)
190 
191  '''
192 
193  if distributed_memory is None:
194  distributed_memory = Config.get().distributed_memory
195 
196  # pylint: disable=too-many-statements, too-many-locals, too-many-branches
197  if api == "":
198  api = Config.get().default_api
199  else:
200  if api not in Config.get().supported_apis:
201  raise GenerationError(
202  "generate: Unsupported API '{0}' specified. Supported "
203  "types are {1}.".format(api, Config.get().supported_apis))
204 
205  # Store Kernel-output options in our Configuration object
206  Config.get().kernel_output_dir = kern_out_path
207  try:
208  Config.get().kernel_naming = kern_naming
209  except ValueError as verr:
210  raise GenerationError("Invalid kernel-renaming scheme supplied: {0}".
211  format(str(verr)))
212 
213  if not os.path.isfile(filename):
214  raise IOError("file '{0}' not found".format(filename))
215  if kernel_path and not os.access(kernel_path, os.R_OK):
216  raise IOError("kernel search path '{0}' not found".format(kernel_path))
217  try:
218  from psyclone.algGen import Alg
219  ast, invoke_info = parse(filename, api=api, invoke_name="invoke",
220  kernel_path=kernel_path,
221  line_length=line_length)
222  psy = PSyFactory(api, distributed_memory=distributed_memory)\
223  .create(invoke_info)
224  if script_name is not None:
225  handle_script(script_name, psy)
226 
227  if api not in API_WITHOUT_ALGORITHM:
228  alg_gen = Alg(ast, psy).gen
229  else:
230  alg_gen = None
231  except Exception:
232  raise
233 
234  return alg_gen, psy.gen

238  '''
239  Parses and checks the command line arguments, calls the generate

246

252  parser = argparse.ArgumentParser(
253  description='Run the PSyclone code generator on a particular file')
254  parser.add_argument('-oalg', help='filename of transformed algorithm code')
255  parser.add_argument(
256  '-opsy', help='filename of generated PSy code')
257  parser.add_argument('-okern',
258  help='directory in which to put transformed kernels, '

261  help='choose a particular api from {0}, '

265  parser.add_argument('filename', help='algorithm-layer source code')
266  parser.add_argument('-s', '--script', help='filename of a PSyclone'
267  ' optimisation script')
268  parser.add_argument(
269  '-d', '--directory', default="", help='path to root of directory '
270  'structure containing kernel source code')
271  # Make the default an empty list so that we can check whether the
272  # user has supplied a value(s) later
273  parser.add_argument(
274  '-I', '--include', default=[], action="append",
275  help='path to Fortran INCLUDE files (nemo API only)')
276  parser.add_argument(
277  '-l', '--limit', dest='limit', action='store_true', default=False,
278  help='limit the fortran line length to 132 characters')
279  parser.add_argument(
280  '-dm', '--dist_mem', dest='dist_mem', action='store_true',
281  help='generate distributed memory code')
282  parser.add_argument(
283  '-nodm', '--no_dist_mem', dest='dist_mem', action='store_false',
284  help='do not generate distributed memory code')
285  parser.add_argument(
286  '--kernel-renaming', default="multiple",
287  choices=configuration.VALID_KERNEL_NAMING_SCHEMES,
288  help="Naming scheme to use when re-naming transformed kernels")
289  parser.add_argument(
290  '--profile', '-p', action="append", choices=Profiler.SUPPORTED_OPTIONS,
291  help="Add profiling hooks for either 'kernels' or 'invokes'")
292  parser.add_argument(
293  '--force-profile', action="append",
294  choices=Profiler.SUPPORTED_OPTIONS,
295  help="Add profiling hooks for either 'kernels' or 'invokes' even if a "
296  "transformation script is used. Use at your own risk.")
297  parser.set_defaults(dist_mem=Config.get().distributed_memory)
298 
299  parser.add_argument("--config", help="Config file with "
300  "PSyclone specific options.")
301  parser.add_argument(
302  '-v', '--version', dest='version', action="store_true",
303  help='Display version information ({0})'.format(__VERSION__))
304 
305  args = parser.parse_args(args)
306 
307  if args.version:
308  print("PSyclone version: {0}".format(__VERSION__))
309 
310  if args.script is not None and args.profile is not None:
311  print("Error: use of automatic profiling in combination with an\n"
312  "optimisation script is not recommened since it may not work\n"
313  "as expected.\n"
314  "You can use --force-profile instead of --profile if you \n"
315  "really want to use both options at the same time.",
316  file=sys.stderr)
317  exit(1)
318 
319  if args.profile is not None and args.force_profile is not None:
320  print("Specify only one of --profile and --force-profile.",
321  file=sys.stderr)
322  exit(1)
323 
324  if args.profile:
325  Profiler.set_options(args.profile)
326  elif args.force_profile:
327  Profiler.set_options(args.force_profile)
328 
329  # If an output directory has been specified for transformed kernels
330  # then check that it is valid
331  if args.okern:
332  if not os.path.exists(args.okern):
333  print("Specified kernel output directory ({0}) does not exist.".
334  format(args.okern), file=sys.stderr)
335  exit(1)
336  if not os.access(args.okern, os.W_OK):
337  print("Cannot write to specified kernel output directory ({0}).".
338  format(args.okern), file=sys.stderr)
339  exit(1)
340  kern_out_path = args.okern
341  else:
342  # We write any transformed kernels to the current working directory
343  kern_out_path = os.getcwd()
344 
345  # If no config file name is specified, args.config is none
346  # and config will load the default config file.
347  Config.get().load(args.config)
348 
349  # Check API, if none is specified, take the setting from the config file
350  if args.api is None:
351  # No command line option, use the one specified in Config - which
352  # is either based on a parameter in the config file, or otherwise
353  # the default:
354  api = Config.get().api
355  elif args.api not in Config.get().supported_apis:
356  print("Unsupported API '{0}' specified. Supported API's are "
357  "{1}.".format(args.api, Config.get().supported_apis),
358  file=sys.stderr)
359  exit(1)
360  else:
361  # There is a valid API specified on the command line. Set it
362  # as API in the config object as well.
363  api = args.api
364  Config.get().api = api
365 
366  # Store the search path(s) for include files
367  if args.include and api != 'nemo':
368  # We only support passing include paths to fparser2 and it's
369  # only the NEMO API that uses fparser2 currently.
370  print("Setting the search path for Fortran include files "
371  "(-I/--include) is only supported for the 'nemo' API.",
372  file=sys.stderr)
373  exit(1)
374 
375  # The Configuration manager checks that the supplied path(s) is/are
376  # valid so protect with a try
377  try:
378  if args.include:
379  Config.get().include_paths = args.include
380  else:
381  # Default is to instruct fparser2 to look in the directory
382  # containing the file being parsed
383  Config.get().include_paths = ["./"]
384  except ConfigurationError as err:
385  print(str(err), file=sys.stderr)
386  exit(1)
387 
388  try:
389  alg, psy = generate(args.filename, api=api,
390  kernel_path=args.directory,
391  script_name=args.script,
392  line_length=args.limit,
393  distributed_memory=args.dist_mem,
394  kern_out_path=kern_out_path,
395  kern_naming=args.kernel_renaming)
396  except NoInvokesError:
397  _, exc_value, _ = sys.exc_info()
398  print("Warning: {0}".format(exc_value))
399  # no invoke calls were found in the algorithm file so we need
400  # not need to process it, or generate any psy layer code so
401  # output the original algorithm file and set the psy file to
402  # be empty
403  alg_file = open(args.filename)
404  alg = alg_file.read()
405  psy = ""
406  except (OSError, IOError, ParseError, GenerationError,
407  RuntimeError):
408  _, exc_value, _ = sys.exc_info()
409  print(exc_value, file=sys.stderr)
410  exit(1)
411  except Exception: # pylint: disable=broad-except
412  print("Error, unexpected exception, please report to the authors:",
413  file=sys.stderr)
414  exc_type, exc_value, exc_tb = sys.exc_info()
415  print("Description ...", file=sys.stderr)
416  print(exc_value, file=sys.stderr)
417  print("Type ...", file=sys.stderr)
418  print(exc_type, file=sys.stderr)
419  print("Stacktrace ...", file=sys.stderr)
420  traceback.print_tb(exc_tb, limit=10, file=sys.stderr)
421  exit(1)
422  if args.limit:
423  fll = FortLineLength()
424  psy_str = fll.process(str(psy))
425  alg_str = fll.process(str(alg))
426  else:
427  psy_str = str(psy)
428  alg_str = str(alg)
429  if args.oalg is not None:
430  my_file = open(args.oalg, "w")
431  my_file.write(alg_str)
432  my_file.close()
433  else:
434  print("Transformed algorithm code:\n%s" % alg_str)
435 
436  if not psy_str:
437  # empty file so do not output anything
438  pass
439  elif args.opsy is not None:
440  my_file = open(args.opsy, "w")
441  my_file.write(psy_str)
442  my_file.close()
443  else:
444  print("Generated psy layer code:\n", psy_str)
