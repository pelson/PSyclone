9 ''' Provides support for breaking long fortran lines into smaller ones

11 for f90 free format is the default)'''

14 def find_break_point(line, max_index, key_list):
15  ''' find the most appropriate break point for a fortran line '''
16 
17  for key in key_list:
18  idx = line.rfind(key, 0, max_index)
19  if idx > 0:
20  return idx+len(key)
21  raise Exception(
22  "Error in find_break_point. No suitable break point found"
23  " for line '" + line[:max_index] + "' and keys '" +
24  str(key_list) + "'")

73  def process(self, fortran_in):
74  ''' takes fortran code as a string as input and output fortran
75  code as a string with any long lines wrapped appropriately '''
76 
77  fortran_out = ""
78  for line in fortran_in.split('\n'):
79  if len(line) > self._line_length:
80  line_type = self._get_line_type(line)
81 
82  c_start = self._cont_start[line_type]
83  c_end = self._cont_end[line_type]
84  key_list = self._key_lists[line_type]
85 
86  break_point = find_break_point(
87  line, self._line_length-len(c_end), key_list)
88  fortran_out += line[:break_point] + c_end + "\n"
89  line = line[break_point:]
90  while len(line) + len(c_start) > self._line_length:
91  break_point = find_break_point(
92  line, self._line_length-len(c_end)-len(c_start),
93  key_list)
94  fortran_out += c_start + line[:break_point] + c_end + "\n"
95  line = line[break_point:]
96  if line:
97  fortran_out += c_start + line + "\n"
98  else:
99  fortran_out += line + "\n"
100 
101  # We add an extra newline so remove it when we return
102  return fortran_out[:-1]
