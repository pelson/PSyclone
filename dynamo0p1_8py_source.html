45 class DynamoPSy(PSy):
46  '''
47  The Dynamo specific PSy class. This creates a Dynamo specific
48  invokes object (which controls all the required invocation calls).
49  Also overrides the PSy gen method so that we generate dynamo
50  specific PSy module code.
51 
52  :param invoke_info: An object containing the required invocation \
53  information for code optimisation and generation.
54  :type invoke_info: :py:class:`psyclone.parse.FileInfo`
55  '''
56  def __init__(self, invoke_info):
57  PSy.__init__(self, invoke_info)
58  self._invokes = DynamoInvokes(invoke_info.calls)
59 
60  @property
61  def gen(self):
62  '''
63  Generate PSy code for the Dynamo0.1 api.
64 
65  :rtype: ast
66 
67  '''
68  from psyclone.f2pygen import ModuleGen, UseGen
69 
70  # create an empty PSy layer module
71  psy_module = ModuleGen(self.name)
72  # include the lfric module
73  lfric_use = UseGen(psy_module, name = "lfric")
74  psy_module.add(lfric_use)
75  # add all invoke specific information
76  self.invokes.gen_code(psy_module)
77  # inline kernel subroutines if requested
78  self.inline(psy_module)
79  return psy_module.root

99  def gen_code(self, parent):
100  ''' Generates Dynamo specific invocation code (the subroutine called
101  by the associated invoke call in the algorithm layer). This
102  consists of the PSy invocation subroutine and the declaration of
103  its arguments.'''
104  from psyclone.f2pygen import SubroutineGen, TypeDeclGen
105  # create the subroutine
106  invoke_sub = SubroutineGen(parent, name = self.name,
107  args = self.psy_unique_var_names)
108  self.schedule.gen_code(invoke_sub)
109  parent.add(invoke_sub)
110  # add the subroutine argument declarations
111  my_typedecl = TypeDeclGen(invoke_sub, datatype = "field_type",
112  entity_decls = self.psy_unique_var_names,
113  intent = "inout")
114  invoke_sub.add(my_typedecl)

125 class DynLoop(Loop):
126  ''' The Dynamo specific Loop class. This passes the Dynamo specific
127  loop information to the base class so it creates the one we require.
128  Creates Dynamo specific loop bounds when the code is being generated.
129  '''
130  def __init__(self, parent=None, loop_type=""):
131  Loop.__init__(self, parent=parent,
132  valid_loop_types=["", "colours", "colour"])
133  self.loop_type = loop_type
134 
135  # Work out the variable name from the loop type
136  if self._loop_type == "colours":
137  self._variable_name = "colour"
138  elif self._loop_type == "colour":
139  self._variable_name = "cell"
140  else:
141  self._variable_name = "cell"
142 
143  def load(self, kern):
144  ''' Load the state of this Loop using the supplied Kernel
145  object. This method is provided so that we can individually
146  construct Loop objects for a given kernel call. '''
147  self._field = kern.arguments.iteration_space_arg()
148  self._field_name = self._field.name
149  self._field_space = self._field.function_space
150 
151  def gen_code(self,parent):
152  ''' Work out the appropriate loop bounds and then call the base
153  class to generate the code '''
154  self._start = "1"
155  if self._loop_type == "colours":
156  self._stop = "ncolour"
157  elif self._loop_type == "colour":
158  self._stop = "ncp_ncolour(colour)"
159  else:
160  self._stop = self.field_name+"%get_ncell()"
161  Loop.gen_code(self,parent)

174 class DynKernCallFactory(object):
175  ''' A Dynamo 0.1 specific kernel call factory. '''
176  @staticmethod
177  def create(call, parent=None):
178 
179  # Loop over cells
180  cloop = DynLoop(parent=parent)
181 
182  # The kernel itself
183  kern = DynKern()
184  kern.load(call, cloop)
185 
186  # Add the kernel as a child of the loop
187  cloop.addchild(kern)
188 
189  # Set-up the loop now we have the kernel object
190  cloop.load(kern)
191 
192  # Return the outermost loop
193  return cloop

210  def gen_code(self, parent):
211  ''' Generates dynamo version 0.1 specific psy code for a call to
212  the dynamo kernel instance. '''
213  from psyclone.f2pygen import CallGen, DeclGen, AssignGen, UseGen
214 
215  # TODO: we simply choose the first field as the lookup for the moment
216  field_name = self.arguments.args[0].name
217 
218  # add a dofmap lookup using first field.
219  # TODO: This needs to be generalised to work for multiple dofmaps
220  parent.add(CallGen(parent, field_name+"%vspace%get_cell_dofmap",
221  ["cell", "map"]))
222  parent.add(DeclGen(parent, datatype = "integer",
223  entity_decls = ["cell"]))
224  parent.add(DeclGen(parent, datatype = "integer", pointer = True,
225  entity_decls = ["map(:)"]))
226 
227  # create the argument list on the fly so we can also create
228  # appropriate variables and lookups
229  arglist = []
230  arglist.append("nlayers")
231  arglist.append("ndf")
232  arglist.append("map")
233 
234  found_gauss_quad = False
235  gauss_quad_arg = None
236  for arg in self._arguments.args:
237  if arg.requires_basis:
238  basis_name = arg.function_space+"_basis_"+arg.name
239  arglist.append(basis_name)
240  new_parent, position = parent.start_parent_loop()
241  new_parent.add(CallGen(new_parent,
242  field_name+"%vspace%get_basis",
243  [basis_name]),
244  position = ["before",
245  position])
246  parent.add(DeclGen(parent, datatype = "real", kind = "dp",
247  pointer = True,
248  entity_decls = [basis_name+"(:,:,:,:,:)"]))
249  if arg.requires_diff_basis:
250  raise GenerationError("differential basis has not yet "
251  "been coded")
252  if arg.requires_gauss_quad:
253  if found_gauss_quad:
254  raise GenerationError("found more than one gaussian "
255  "quadrature in this kernel")
256  found_gauss_quad = True
257  gauss_quad_arg = arg
258  dataref = "%data"
259  arglist.append(arg.name+dataref)
260 
261  if found_gauss_quad:
262  gq_name = "gaussian_quadrature"
263  arglist.append(gauss_quad_arg.name+"%"+gq_name)
264 
265  # generate the kernel call and associated use statement
266  parent.add(CallGen(parent, self._name, arglist))
267  if not self.module_inline:
268  parent.add(UseGen(parent, name=self._module_name,
269  only=True, funcnames=[self._name]))
270 
271  # declare and initialise the number of layers and the number
272  # of degrees of freedom. Needs to be generalised.
273  parent.add(DeclGen(parent, datatype = "integer",
274  entity_decls = ["nlayers", "ndf"]))
275  new_parent, position = parent.start_parent_loop()
276  new_parent.add(AssignGen(new_parent, lhs = "nlayers",
277  rhs = field_name+"%get_nlayers()"),
278  position = ["before", position])
279  new_parent.add(AssignGen(new_parent, lhs = "ndf",
280  rhs = field_name+"%vspace%get_ndf()"),
281  position = ["before", position])
