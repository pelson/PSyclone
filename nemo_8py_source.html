82  def _create_child(self, child, parent=None):
83  '''
84  Adds Nemo API specific processors for certain fparser2 types
85  before calling the parent _create_child.
86 
87  :param child: node in fparser2 AST.
88  :type child: :py:class:`fparser.two.utils.Base`
89  :param parent: Parent node in the PSyclone IR we are constructing.
90  :type parent: :py:class:`psyclone.psyGen.Node`

92  :rtype: :py:class:`psyclone.psyGen.Node`

94  if isinstance(child, Fortran2003.Block_Nonlabel_Do_Construct):
95  return NemoLoop(child, parent=parent)

100  elif NemoIfBlock.match(child):
101  return NemoIfBlock(child, parent=parent)
102  else:
103  return super(NemoFparser2ASTProcessor, self)._create_child(child)

106 class NemoInvoke(Invoke):
107  '''
108  Represents a NEMO 'Invoke' which, since NEMO is existing code, means
109  an existing program unit, e.g. a subroutine.
110 
111  :param ast: node in fparser2 AST representing the program unit.
112  :type ast: :py:class:`fparser.two.Fortran2003.Main_Program` or \
113  :py:class:`fparser.two.Fortran2003.Module`
114  :param str name: the name of this Invoke (program unit).
115  '''
116  def __init__(self, ast, name):
117  self._schedule = None
118  self._name = name
119  # Store the whole fparser2 AST
120  self._ast = ast
121  self._name_space_manager = NameSpaceFactory().create()
122  from fparser.two.Fortran2003 import Execution_Part, Specification_Part
123 
124  # Find the section of the tree containing the execution part
125  # of the code
126  exe_part = get_child(ast, Execution_Part)
127  if not exe_part:
128  # This subroutine has no execution part so we skip it
129  # TODO log this event
130  return
131 
132  # Store the root of this routine's specification in the AST
133  self._spec_part = get_child(ast, Specification_Part)
134 
135  # We now walk through the AST produced by fparser2 and construct a
136  # new AST using objects from the nemo module.
137  self._schedule = NemoSchedule(self, exe_part)
138 
139  def update(self):
140  '''Updates the fparser2 AST associated with this Schedule to make it
141  reflect any transformations that have been applied to the
142  PSyclone AST.
143 
144  '''
145  if not self._schedule:
146  return
147  self._schedule.update()

150 class NemoInvokes(Invokes):
151  '''
152  Class capturing information on all 'Invokes' (program units) within
153  a single NEMO source file.
154 
155  :param ast: The fparser2 AST for the whole Fortran source file
156  :type ast: :py:class:`fparser.two.Fortran2003.Main_Program`
157  '''
158  def __init__(self, ast):
159  from fparser.two.Fortran2003 import Main_Program, Program_Stmt, \
160  Subroutine_Subprogram, Function_Subprogram, Function_Stmt, \
161  Subroutine_Stmt, Name
162 
163  self.invoke_map = {}
164  self.invoke_list = []
165  # Keep a pointer to the whole fparser2 AST
166  self._ast = ast
167 
168  # Find all the subroutines contained in the file
169  routines = walk_ast(ast.content, [Subroutine_Subprogram,
170  Function_Subprogram])
171  # Add the main program as a routine to analyse - take care
172  # here as the Fortran source file might not contain a
173  # main program (might just be a subroutine in a module)
174  main_prog = get_child(ast, Main_Program)
175  if main_prog:
176  routines.append(main_prog)
177 
178  # Analyse each routine we've found
179  for subroutine in routines:
180  # Get the name of this (sub)routine
181  substmt = walk_ast(subroutine.content,
182  [Subroutine_Stmt, Function_Stmt, Program_Stmt])
183  if isinstance(substmt[0], Function_Stmt):
184  for item in substmt[0].items:
185  if isinstance(item, Name):
186  sub_name = str(item)
187  break
188  else:
189  sub_name = str(substmt[0].get_name())
190 
191  my_invoke = NemoInvoke(subroutine, name=sub_name)
192  self.invoke_map[sub_name] = my_invoke
193  self.invoke_list.append(my_invoke)
194 
195  def update(self):
196  ''' Walk down the tree and update the underlying fparser2 AST
197  to reflect any transformations. '''
198  for invoke in self.invoke_list:
199  invoke.update()

338  @staticmethod
339  def match(node):
340  '''
341  Whether or not the AST fragment pointed to by node represents a
342  kernel. A kernel is defined as a section of code that sits
343  within a recognised loop structure and does not itself contain
344  loops or IO operations.
345 
346  :param node: Node in fparser2 AST to check.
347  :type node: :py:class:`fparser.two.Fortran2003.Base`
348  :returns: True if this node conforms to the rules for a kernel.
349  :rtype: bool
350  '''
351  from fparser.two.Fortran2003 import Subscript_Triplet, \
352  Block_Nonlabel_Do_Construct, Write_Stmt, Read_Stmt
353  child_loops = walk_ast(node.content,
354  [Block_Nonlabel_Do_Construct, Write_Stmt,
355  Read_Stmt])
356  if child_loops:
357  # A kernel cannot contain other loops or reads or writes
358  return False
359 
360  # Currently a kernel cannot contain implicit loops.
361  # TODO we may have to differentiate between implicit loops over
362  # grid points and any other implicit loop. Possibly using the
363  # scope of the array being accessed?
364  impl_loops = walk_ast(node.content, [Subscript_Triplet])
365  if impl_loops:
366  return False
367 
368  return True

491 class NemoLoop(Loop, NemoFparser2ASTProcessor):
492  '''
493  Class representing a Loop in NEMO.
494 
495  :param ast: node in the fparser2 AST representing the loop.
496  :type ast: :py:class:`fparser.two.Block_Nonlabel_Do_Construct`
497  :param parent: parent of this NemoLoop in the PSyclone AST.
498  :type parent: :py:class:`psyclone.psyGen.Node`
499  '''
500  def __init__(self, ast, parent=None):
501  from fparser.two.Fortran2003 import Loop_Control
502  Loop.__init__(self, parent=parent,
503  valid_loop_types=VALID_LOOP_TYPES)
504  NemoFparser2ASTProcessor.__init__(self)
505  # Keep a ptr to the corresponding node in the AST
506  self._ast = ast
507 
508  # Get the loop variable
509  ctrl = walk_ast(ast.content, [Loop_Control])
510  # Second element of items member of Loop Control is itself a tuple
511  # containing:
512  # Loop variable, [start value expression, end value expression, step
513  # expression]
514  # Loop variable will be an instance of Fortran2003.Name
515  loop_var = str(ctrl[0].items[1][0])
516  self._variable_name = str(loop_var)
517 
518  # Identify the type of loop
519  if self._variable_name in NEMO_LOOP_TYPE_MAPPING:
520  self.loop_type = NEMO_LOOP_TYPE_MAPPING[self._variable_name]
521  else:
522  self.loop_type = "unknown"
523 
524  # Get the loop limits. These are given in a list which is the second
525  # element of a tuple which is itself the second element of the items
526  # tuple:
527  # (None, (Name('jk'), [Int_Literal_Constant('1', None), Name('jpk'),
528  # Int_Literal_Constant('1', None)]), None)
529  limits_list = ctrl[0].items[1][1]
530  self._start = str(limits_list[0])
531  self._stop = str(limits_list[1])
532  if len(limits_list) == 3:
533  self._step = str(limits_list[2])
534  else:
535  # Default loop increment is 1
536  self._step = "1"
537 
538  # Is this loop body a kernel?
539  if NemoKern.match(self._ast):
540  self.addchild(NemoKern(self._ast, parent=self))
541  return
542  # It's not - walk on down the AST...
543  self.process_nodes(self, self._ast.content, self._ast)
544 
545  def __str__(self):
546  result = ("NemoLoop[" + self._loop_type + "]: " + self._variable_name +
547  "=" + ",".join([self._start, self._stop, self._step]) + "\n")
548  for entity in self._children:
549  result += str(entity) + "\n"
550  result += "EndLoop"
551  return result

575 class NemoImplicitLoop(NemoLoop):
576  '''
577  Class representing an implicit loop in NEMO (i.e. using Fortran array
578  syntax).
579 
580  :param ast: the part of the fparser2 AST representing the loop.
581  :type ast: :py:class:`fparser.two.Fortran2003.Assignment_Stmt`
582  :param parent: the parent of this Loop in the PSyIRe.
583  :type parent: :py:class:`psyclone.psyGen.Node`
584 
585  :raises NotImplementedError: if the array slice has explicit bounds.
586  :raises GenerationError: if an array slice is not in dimensions 1-3 of \
587  the array.
588  '''
589  def __init__(self, ast, parent=None):
590  from fparser.common.readfortran import FortranStringReader
591  Loop.__init__(self, parent=parent,
592  valid_loop_types=VALID_LOOP_TYPES)
593  # Keep a ptr to the corresponding node in the AST
594  self._ast = ast
595 
596  # Get a reference to the name-space manager
597  name_space_manager = NameSpaceFactory().create()
598 
599  # Find all uses of array syntax in the statement
600  subsections = walk_ast(ast.items, [Fortran2003.Section_Subscript_List])
601  # A Section_Subscript_List is a tuple with each item the
602  # array-index expressions for the corresponding dimension of the array.
603  for idx, item in enumerate(subsections[0].items):
604  if isinstance(item, Fortran2003.Subscript_Triplet):
605  # A Subscript_Triplet has a 3-tuple containing the expressions
606  # for the start, end and increment of the slice. If any of
607  # these are not None then we have an explicit range of some
608  # sort and we do not yet support that.
609  # TODO allow for implicit loops with specified bounds
610  # (e.g. 2:jpjm1)
611  if [part for part in item.items if part]:
612  raise NotImplementedError(
613  "Support for implicit loops with specified bounds is "
614  "not yet implemented: '{0}'".format(str(ast)))
615  # If an array index is a Subscript_Triplet then it is a range
616  # and thus we need to create an explicit loop for this
617  # dimension.
618  outermost_dim = idx
619 
620  if outermost_dim < 0 or outermost_dim > 2:
621  raise GenerationError(
622  "Array section in unsupported dimension ({0}) for code "
623  "'{1}'".format(outermost_dim+1, str(ast)))
624 
625  self._step = 1
626  # TODO ensure no name clash is possible with variables that
627  # already exist in the NEMO source.
628  self.loop_type = NEMO_INDEX_ORDERING[outermost_dim]
629  self._start = VALID_LOOP_TYPES[self.loop_type]["start"]
630  self._stop = VALID_LOOP_TYPES[self.loop_type]["stop"]
631  var_name = "psy_" + VALID_LOOP_TYPES[self.loop_type]["var"]
632  self._variable_name = name_space_manager.create_name(
633  root_name=var_name, context="PSyVars", label=var_name)
634 
635  # TODO Since the fparser2 AST does not have parent
636  # information (and no other way of getting to the root node), it is
637  # currently not possible to insert a declaration in the correct
638  # location.
639  # For the moment, we can work around the fparser2 AST limitation
640  # by using the fact that we *can* get hold of the PSyclone Invoke
641  # object and that contains a reference to the root of the fparser2
642  # AST...
643  name = Fortran2003.Name(FortranStringReader(self._variable_name))
644  prog_unit = self.root.invoke._ast
645  spec = walk_ast(prog_unit.content, [Fortran2003.Specification_Part])
646  if not spec:
647  names = walk_ast(prog_unit.content, [Fortran2003.Name])
648  # TODO create a Specification_Part
649  raise InternalError("No specification part found for routine {0}!".
650  format(names[0]))
651  # TODO check that this variable has not already been declared
652  # Requires that we capture all variable declarations in the routine
653  # in some sort of management class.
654  decln = Fortran2003.Type_Declaration_Stmt(
655  FortranStringReader("integer :: {0}".format(self._variable_name)))
656  spec[0].content.append(decln)
657 
658  for subsec in subsections:
659  # A tuple is immutable so work with a list
660  indices = list(subsec.items)
661  # Replace the colon with our new variable name
662  indices[outermost_dim] = name
663  # Replace the original tuple with a new one
664  subsec.items = tuple(indices)
665 
666  # Create an explicit loop
667  text = ("do {0}=1,{1},{2}\n"
668  " replace = me\n"
669  "end do\n".format(self._variable_name, self._stop, self._step))
670  loop = Fortran2003.Block_Nonlabel_Do_Construct(
671  FortranStringReader(text))
672  parent_index = ast._parent.content.index(ast)
673  # Insert it in the fparser2 AST at the location of the implicit
674  # loop
675  ast._parent.content.insert(parent_index, loop)
676  # Replace the content of the loop with the (modified) implicit
677  # loop
678  loop.content[1] = ast
679  # Remove the implicit loop from its original parent in the AST
680  ast._parent.content.remove(ast)
681  # Update the parent of the AST
682  ast._parent = loop
683  # Update our own pointer into the AST to now point to the explicit
684  # loop we've just created
685  self._ast = loop
686 
687  if NemoImplicitLoop.match(ast):
688  # We still have an implicit loop so recurse
689  self.addchild(NemoImplicitLoop(ast, parent=self))
690  else:
691  # We must be left with a kernel
692  self.addchild(NemoKern(ast, parent=self))

717 class NemoIfBlock(IfBlock, NemoFparser2ASTProcessor):
718  '''
719  Represents an if-block within a NEMO Schedule.
720  Within the fparser2 AST, an if-block is represented as:
721  If_Then_Stmt
722  statement(s)
723  Else_Stmt
724  further statement(s)
725  End_If_Stmt
726  i.e. the statements contained inside the if-block are siblings
727  of the control statements, not children of them.
728 
729  :param ast: reference to fparser2 AST representing if block.
730  :type ast: :py:class:`fparser.two.Fortran2003.If_Construct`
731  :param parent: parent node of this if block in the PSyIRe.
732  :type parent: :py:class:`psyclone.psyGen.Node`
733 
734  :raises InternalError: if the fparser2 AST does not have the expected \
735  structure.
736  '''
737  def __init__(self, ast, parent=None):
738  super(NemoIfBlock, self).__init__(parent=parent)
739  NemoFparser2ASTProcessor.__init__(self)
740  # Keep a ptr to the corresponding node in the AST
741  self._ast = ast
742  # Check that the fparser2 AST has the expected structure
743  if not isinstance(ast.content[0], Fortran2003.If_Then_Stmt):
744  raise InternalError("Failed to find opening if then statement: "
745  "{0}".format(str(ast)))
746  if not isinstance(ast.content[-1], Fortran2003.End_If_Stmt):
747  raise InternalError("Failed to find closing end if statement: "
748  "{0}".format(str(ast)))
749  clause_indices = []
750  for idx, child in enumerate(ast.content):
751  child._parent = self._ast # Retrofit parent info
752  if isinstance(child, (Fortran2003.If_Then_Stmt,
753  Fortran2003.Else_Stmt,
754  Fortran2003.Else_If_Stmt,
755  Fortran2003.End_If_Stmt)):
756  clause_indices.append(idx)
757  # Create the body of the main If
758  end_idx = clause_indices[1]
759  self._condition = str(ast.content[0].items[0])
760  self.process_nodes(parent=self,
761  nodes=ast.content[1:end_idx],
762  nodes_parent=ast)
763  # Now deal with any other clauses (i.e. "else if" or "else")
764  # An If block has one fewer clauses than it has control statements
765  # (c.f. panels and posts):
766  num_clauses = len(clause_indices) - 1
767  for idx in range(1, num_clauses):
768  start_idx = clause_indices[idx]
769  # No need to subtract 1 here as Python's slice notation means
770  # that the end_idx'th element is excluded
771  end_idx = clause_indices[idx+1]
772  ast.content[start_idx]._parent = ast # Retrofit parent info
773  self.addchild(NemoIfClause(ast.content[start_idx:end_idx],
774  parent=self))
