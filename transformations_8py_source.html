67 class RegionTrans(Transformation):
68  '''This class is a base class for all transforms that act on list of
69  nodes. It gives access to a _validate function that makes sure that
70  the nodes in the list are in the same order as in the original AST,
71  no node is duplicated, and that all nodes have the same parent.
72  '''
73 
74  # Avoid pylint warning about abstract functions (apply, name) not
75  # overwritten:
76  # pylint: disable=abstract-method,arguments-differ

78  def _validate(self, node_list):
79  '''Test if the nodes in node_list are in the original order.
80 
81  :param list node_list: List of nodes.
82  :raises TransformationError: If the nodes in the list are not\
83  in the original order in which they are in the AST,\
84  a node is duplicated or the nodes have different parents.
85  '''
86 
87  node_parent = node_list[0].parent
88  prev_position = -1
89  for child in node_list:
90  if child.parent is not node_parent:
91  raise TransformationError(
92  "Error in {0} transformation: supplied nodes "
93  "are not children of the same Schedule/parent."
94  .format(self.name))
95  if prev_position >= 0 and prev_position+1 != child.position:
96  raise TransformationError(
97  "Children are not consecutive children of one parent: "
98  "child '{0}' has position {1}, but previous child had "
99  "position {2}."
100  .format(str(child), child.position, prev_position))
101  prev_position = child.position

134 class LoopFuseTrans(Transformation):
135  ''' Provides a loop-fuse transformation.
136  For example:
137 
138  >>> from psyclone.parse import parse
139  >>> from psyclone.psyGen import PSyFactory
140  >>> ast,invokeInfo=parse("dynamo.F90")
141  >>> psy=PSyFactory("dynamo0.1").create(invokeInfo)
142  >>> schedule=psy.invokes.get('invoke_v3_kernel_type').schedule
143  >>> schedule.view()
144  >>>
145  >>> from psyclone.transformations import LoopFuseTrans
146  >>> trans=LoopFuseTrans()
147  >>> new_schedule,memento=trans.apply(schedule.children[0],
148  schedule.children[1])
149  >>> new_schedule.view()
150  '''
151 
152  def __str__(self):
153  return "Fuse two adjacent loops together"
154 
155  @property
156  def name(self):
157  ''' Returns the name of this transformation as a string.'''
158  return "LoopFuse"
159 
160  def _validate(self, node1, node2):
161  '''Perform various checks to ensure that it is valid to apply the
162  LoopFuseTrans transformation to the supplied nodes
163 
164  :param node1: the first node we are checking
165  :type node1: :py:class:`psyclone.psyGen.Node`
166  :param node2: the second node we are checking
167  :type node2: :py:class:`psyclone.psyGen.Node`
168  :raises TransformationError: if one or both of the nodes is/are not a
169  :py:class:`psyclone.psyGen.Loop`
170  :raises TransformationError: if the nodes do not have the same parent
171  :raises TransformationError: if the nodes are not next to each
172  other in the tree
173  :raises TransformationError: if the
174  :py:class:`psyclone.psyGen.Loop`s do not have the same
175  iteration space
176  '''
177 
178  # Check that the supplied Node is a Loop
179  from psyclone.psyGen import Loop
180  if not isinstance(node1, Loop) or not isinstance(node2, Loop):
181  raise TransformationError("Error in LoopFuse transformation. "
182  "At least one of the nodes is not "
183  "a loop")
184  # check loop1 and loop2 have the same parent
185  if not node1.sameParent(node2):
186  raise TransformationError("Error in LoopFuse transformation. "
187  "loops do not have the same parent")
188  # check node1 and node2 are next to each other
189  if abs(node1.position-node2.position) != 1:
190  raise TransformationError("Error in LoopFuse transformation. "
191  "nodes are not siblings who are "
192  "next to each other")
193  # Check iteration space is the same
194  if node1.iteration_space != node2.iteration_space:
195  raise TransformationError("Error in LoopFuse transformation. "
196  "Loops do not have the same "
197  "iteration space")
198 
199  def apply(self, node1, node2):
200  '''Fuse the loops represented by :py:obj:`node1` and
201  :py:obj:`node2`.'''
202 
203  self._validate(node1, node2)
204 
205  schedule = node1.root
206 
207  # create a memento of the schedule and the proposed transformation
208  from psyclone.undoredo import Memento
209  keep = Memento(schedule, self, [node1, node2])
210 
211  # add loop contents of node2 to node1
212  node1.children.extend(node2.children)
213 
214  # change the parent of the loop contents of node2 to node1
215  for child in node2.children:
216  child.parent = node1
217 
218  # remove node2
219  node2.parent.children.remove(node2)
220 
221  return schedule, keep

270  '''Performs error checking before calling the
271  :py:meth:`~LoopFuseTrans.apply` method of the
272  :py:class:`base class <LoopFuseTrans>` in order
273  to fuse two Dynamo loops.
274 
275  '''
276 
277  def __str__(self):
278  return ("Fuse two adjacent loops together with Dynamo-specific "

281  @property
282  def name(self):
283  ''' Returns the name of this transformation as a string.'''

286  def apply(self, node1, node2, same_space=False):
287  '''
288  Fuse the two Dynamo loops represented by :py:obj:`node1` and
289  :py:obj:`node2`. The optional same_space flag asserts that an
290  unknown iteration space (i.e. any_space) matches the other
291  iteration space. This is set at the users own risk.
292 
293  :param node1: First Loop to fuse.
294  :type node1: :py:class:`psyclone.dynamo0p3.DynLoop`
295  :param node2: Second Loop to fuse.
296  :type node2: :py:class:`psyclone.dynamo0p3.DynLoop`
297  :returns: two-tuple of modified schedule and Memento
298  :rtype: :py:class:`psyclone.psyGen.Schedule`, \
299  :py:class:`psyclone.undoredo.Memento`
300  :raises TransformationError: if either of the supplied loops contains \
301  an inter-grid kernel.
302  '''
303  LoopFuseTrans._validate(self, node1, node2)
304 
305  # Check that we don't have an inter-grid kernel
306  check_intergrid(node1)
307  check_intergrid(node2)
308 
309  from psyclone.dynamo0p3 import VALID_FUNCTION_SPACES
310  try:
311  if node1.field_space.orig_name in VALID_FUNCTION_SPACES and \
312  node2.field_space.orig_name in VALID_FUNCTION_SPACES:
313  if node1.field_space.orig_name != node2.field_space.orig_name:
314  if same_space:
315  info = (
316  " Note, the same_space flag was set, but "
317  "does not apply because neither field is "
318  "ANY_SPACE.")
319  else:
320  info = ""
321  raise TransformationError(
322  "Error in DynamoLoopFuse transformation. "
323  "Cannot fuse loops that are over different spaces: "
324  "{0} {1}.{2}".format(node1.field_space.orig_name,
325  node2.field_space.orig_name,
326  info))
327  else: # one or more of the function spaces is any_space
328  if not same_space:
329  raise TransformationError(
330  "DynamoLoopFuseTrans. One or more of the iteration "
331  "spaces is unknown ('any_space') so loop fusion might "
332  "be invalid. If you know the spaces are the same then "
333  "please set the 'same_space' optional argument to "
334  "True.")
335  if node1.upper_bound_name != node2.upper_bound_name:
336  raise TransformationError(
337  "Error in DynamoLoopFuse transformation. The upper bound "
338  "names are not the same. Found '{0}' and '{1}'".
339  format(node1.upper_bound_name, node2.upper_bound_name))
340  if node1.upper_bound_halo_depth != node2.upper_bound_halo_depth:
341  raise TransformationError(
342  "Error in DynamoLoopFuse transformation. The halo-depth "
343  "indices are not the same. Found '{0}' and '{1}'".
344  format(node1.upper_bound_halo_depth,
345  node2.upper_bound_halo_depth))
346  from psyclone.psyGen import MAPPING_SCALARS, MAPPING_REDUCTIONS
347  arg_types = MAPPING_SCALARS.values()
348  arg_accesses = MAPPING_REDUCTIONS.values()
349  node1_red_args = node1.args_filter(arg_types=arg_types,
350  arg_accesses=arg_accesses)
351  node2_red_args = node2.args_filter(arg_types=arg_types,
352  arg_accesses=arg_accesses)
353 
354  if node1_red_args and node2_red_args:
355  raise TransformationError(
356  "Error in DynamoLoopFuse transformation. "
357  "Cannot fuse loops when each loop already "
358  "contains a reduction")
359 
360  if node1_red_args:
361  for reduction_arg in node1_red_args:
362  other_args = node2.args_filter()
363  for arg in other_args:
364  if reduction_arg.name == arg.name:
365  raise TransformationError(
366  "Error in DynamoLoopFuse transformation. "
367  "Cannot fuse loops as the first loop "
368  "has a reduction and the second loop "
369  "reads the result of the reduction")
370 
371  return LoopFuseTrans.apply(self, node1, node2)
372  except TransformationError as err:
373  raise err
374  except Exception as err:
375  raise TransformationError("Unexpected exception: {0}".
376  format(err))

379 @six.add_metaclass(abc.ABCMeta)
380 class ParallelLoopTrans(Transformation):
381 
382  '''
383  Adds an orphaned directive to a loop indicating that it should be
384  parallelised. i.e. the directive must be inside the scope of some
385  other Parallel REGION. This condition is tested at
386  code-generation time.
387 
388  '''
389  @abc.abstractmethod
390  def __str__(self):
391  return # pragma: no cover
392 
393  @abc.abstractproperty
394  def name(self):
395  '''
396  :returns: the name of this transformation as a string.
397  :rtype: str
398  '''

400  @abc.abstractmethod
401  def _directive(self, parent, children, collapse=None):
402  '''
403  Returns the directive object to insert into the Schedule.
404  Must be implemented by sub-class.
405 
406  :param parent: the parent of this Directive in the Schedule.
407  :type parent: :py:class:`psyclone.psyGen.Node`.
408  :param children: list of nodes that will be children of this Directive.
409  :type children: list of :py:class:`psyclone.psyGen.Node`.
410  :param int collapse: the number of tightly-nested loops to which \
411  this directive applies or None.
412  :returns: the new Directive node.
413  :rtype: sub-class of :py:class:`psyclone.psyGen.Directive`.
414  '''
415 
416  def _validate(self, node, collapse=None):
417  '''
418  Perform validation checks before applying the transformation
419 
420  :param node: the node we are checking.
421  :type node: :py:class:`psyclone.psyGen.Node`.
422  :param int collapse: number of nested loops to collapse or None.
423  :raises TransformationError: if the node is not a \
424  :py:class:`psyclone.psyGen.Loop`
425  :raises TransformationError: if the \
426  :py:class:`psyclone.psyGen.Loop` loop iterates over colours
427 
428  '''
429  # Check that the supplied node is a Loop
430  from psyclone.psyGen import Loop
431  if not isinstance(node, Loop):
432  raise TransformationError(
433  "Cannot apply a parallel-loop directive to something that is "
434  "not a loop")
435  # Check we are not a sequential loop
436  # TODO add a list of loop types that are sequential
437  if node.loop_type == 'colours':
438  raise TransformationError("Error in "+self.name+" transformation. "
439  "The target loop is over colours and "
440  "must be computed serially.")
441  # If 'collapse' is specified, check that it is an int and that the
442  # loop nest has at least that number of loops in it
443  if collapse:
444  if not isinstance(collapse, int):
445  raise TransformationError(
446  "The 'collapse' argument must be an integer but got an "
447  "object of type {0}".format(type(collapse)))
448  if collapse < 2:
449  raise TransformationError(
450  "It only makes sense to collapse 2 or more loops "
451  "but got a value of {0}".format(collapse))
452  # Count the number of loops in the loop nest
453  loop_count = 0
454  cnode = node
455  while isinstance(cnode, Loop):
456  loop_count += 1
457  # Loops must be tightly nested (no intervening statements)
458  cnode = cnode.children[0]
459  if collapse > loop_count:
460  raise TransformationError(
461  "Cannot apply COLLAPSE({0}) clause to a loop nest "
462  "containing only {1} loops".format(collapse, loop_count))
463 
464  def apply(self, node, collapse=None):
465  '''
466  Apply the Loop transformation to the specified node in a
467  Schedule. This node must be a Loop since this transformation
468  corresponds to wrapping the generated code with directives,
469  e.g. for OpenMP:
470 
471  .. code-block:: fortran
472 
473  !$OMP DO
474  do ...
475  ...
476  end do
477  !$OMP END DO
478 
479  At code-generation time (when gen_code()` is called), this node must be
480  within (i.e. a child of) a PARALLEL region.
481 
482  :param node: the supplied node to which we will apply the \
483  Loop transformation.
484  :type node: :py:class:`psyclone.psyGen.Node`.
485  :param int collapse: number of loops to collapse into single \
486  iteration space or None.
487  :return: (:py:class:`psyclone.psyGen.Schedule`, \
488  :py:class:`psyclone.undoredo.Memento`)
489 
490  '''
491  self._validate(node, collapse)
492 
493  schedule = node.root
494 
495  # create a memento of the schedule and the proposed
496  # transformation
497  from psyclone.undoredo import Memento
498  keep = Memento(schedule, self, [node, collapse])
499 
500  # keep a reference to the node's original parent and its index as these
501  # are required and will change when we change the node's location
502  node_parent = node.parent
503  node_position = node.position
504 
505  # Add our orphan loop directive setting its parent to the node's
506  # parent and its children to the node. This calls down to the sub-class
507  # to get the type of directive we require.
508  directive = self._directive(node_parent, [node], collapse)
509 
510  # Add the loop directive as a child of the node's parent
511  node_parent.addchild(directive, index=node_position)
512 
513  # Change the node's parent to be the loop directive.
514  node.parent = directive
515 
516  # Remove the reference to the loop from the original parent.
517  node_parent.children.remove(node)
518 
519  return schedule, keep

522 class OMPLoopTrans(ParallelLoopTrans):
523 
524  '''
525  Adds an orphaned OpenMP directive to a loop. i.e. the directive
526  must be inside the scope of some other OMP Parallel
527  REGION. This condition is tested at code-generation time. The
528  optional 'reprod' argument in the apply method decides whether
529  standard OpenMP reduction support is to be used (which is not
530  reproducible) or whether a manual reproducible reproduction is
531  to be used.
532 
533  :param str omp_schedule: The OpenMP schedule to use.
534 
535  For example:
536 
537  >>> from psyclone.parse import parse, ParseError
538  >>> from psyclone.psyGen import PSyFactory, GenerationError
539  >>> api = "gocean1.0"
540  >>> filename = "nemolite2d_alg.f90"
541  >>> ast, invokeInfo = parse(filename, api=api, invoke_name="invoke")
542  >>> psy = PSyFactory(api).create(invokeInfo)
543  >>> print psy.invokes.names
544  >>>
545  >>> from psyclone.psyGen import TransInfo
546  >>> t = TransInfo()
547  >>> ltrans = t.get_trans_name('OMPLoopTrans')
548  >>> rtrans = t.get_trans_name('OMPParallelTrans')
549  >>>
550  >>> schedule = psy.invokes.get('invoke_0').schedule
551  >>> schedule.view()
552  >>> new_schedule = schedule
553  >>>
554  # Apply the OpenMP Loop transformation to *every* loop
555  # in the schedule
556  >>> for child in schedule.children:
557  >>> newschedule, memento = ltrans.apply(child, reprod=True)
558  >>> schedule = newschedule
559  >>>
560  # Enclose all of these loops within a single OpenMP
561  # PARALLEL region
562  >>> rtrans.omp_schedule("dynamic,1")
563  >>> newschedule, memento = rtrans.apply(schedule.children)
564  >>>
565  >>>
566 
567  '''
568  def __init__(self, omp_schedule="static"):
569  # Whether or not to generate code for (run-to-run on n threads)
570  # reproducible OpenMP reductions. This setting can be overridden
571  # via the `reprod` argument to the apply() method.
572  self._reprod = Config.get().reproducible_reductions
573 
574  self._omp_schedule = ""
575  # Although we create the _omp_schedule attribute above (so that
576  # pylint doesn't complain), we actually set its value using
577  # the setter method in order to make use of the latter's error
578  # checking.
579  self.omp_schedule = omp_schedule
580 
581  super(OMPLoopTrans, self).__init__()
582

584  return "Adds an 'OpenMP DO' directive to a loop"

589  return "OMPLoopTrans"
