116 def get_stencil(metadata, valid_types):
117  '''
118  Returns stencil_type and stencil_extent as a dictionary
119  object from stencil metadata if the metadata conforms to the
120  stencil(type[,extent]) format
121 
122  :param metadata: Component of kernel meta-data stored as a node in the
123  parser AST
124  :type metadata: :py:class:`psyclone.expression.FunctionVar`
125  :param list valid_types: List of valid stencil types (strings)
126  :return: The stencil type and extent described in the meta-data
127  :rtype: dict with keys 'type' (str) and 'extent' (int)
128 
129  :raises ParseError: if the supplied meta-data is not a recognised
130  stencil specification
131  '''
132 
133  if not isinstance(metadata, expr.FunctionVar):
134  raise ParseError(
135  "Expecting format stencil(<type>[,<extent>]) but found the "
136  "literal {0}".format(metadata))
137  if metadata.name.lower() != "stencil" or not metadata.args:
138  raise ParseError(
139  "Expecting format stencil(<type>[,<extent>]) but found {0}".
140  format(metadata))
141  if len(metadata.args) > 2:
142  raise ParseError(
143  "Expecting format stencil(<type>[,<extent>]) but there must "
144  "be at most two arguments inside the brackets {0}".
145  format(metadata))
146  if not isinstance(metadata.args[0], expr.FunctionVar):
147  if isinstance(metadata.args[0], str):
148  raise ParseError(
149  "Expecting format stencil(<type>[,<extent>]). However, "
150  "the specified <type> '{0}' is a literal and therefore is "
151  "not one of the valid types '{1}'".
152  format(metadata.args[0], valid_types))
153  else:
154  raise ParseError(
155  "Internal error, expecting either FunctionVar or "
156  "str from the expression analyser but found {0}".
157  format(type(metadata.args[0])))
158  if metadata.args[0].args:
159  raise ParseError(
160  "Expected format stencil(<type>[,<extent>]). However, the "
161  "specified <type> '{0}' includes brackets")
162  stencil_type = metadata.args[0].name
163  if stencil_type not in valid_types:
164  raise ParseError(
165  "Expected format stencil(<type>[,<extent>]). However, the "
166  "specified <type> '{0}' is not one of the valid types '{1}'".
167  format(stencil_type, valid_types))
168 
169  stencil_extent = None
170  if len(metadata.args) == 2:
171  if not isinstance(metadata.args[1], str):
172  raise ParseError(
173  "Expected format stencil(<type>[,<extent>]). However, the "
174  "specified <extent> '{0}' is not an integer".
175  format(metadata.args[1]))
176  stencil_extent = int(metadata.args[1])
177  if stencil_extent < 1:
178  raise ParseError(
179  "Expected format stencil(<type>[,<extent>]). However, the "
180  "specified <extent> '{0}' is less than 1".
181  format(str(stencil_extent)))
182  raise ParseError(
183  "Kernels with fixed stencil extents are not currently "
184  "supported")
185  return {"type": stencil_type, "extent": stencil_extent}

285 class Element(object):
286  @staticmethod
287  def unpack(string_or_expr):
288  if isinstance(string_or_expr, str):
289  p = expr.FORT_EXPRESSION.parseString(string_or_expr)[0]
290  else:
291  p = string_or_expr
292  if isinstance(p, expr.Grouping):
293  p = p.expr
294  if isinstance(p, expr.BinaryOperator):
295  assert all(a == '*' for a in p.symbols)
296  eles = p.operands
297  else:
298  assert isinstance(p, expr.FunctionVar)
299  eles = [p]
300  order = eles[0].args[0] if eles[0].args else None
301  ele = Element(eles[0].name, order)
302  for e in eles[1:]:
303  order = e.args[0] if e.args else None
304  ele *= Element(e.name, order)
305  return ele

349  An elemental Kernel procedure.

352  :type ktype_ast: :py:class:`fparser.one.block_statements.Type`
353  :param str ktype_name: name of the Fortran type holding the Kernel \
354  meta-data.
355  :param modast: the fparser1 parse tree for the module containing the \
356  Kernel routine.
357  :type modast: :py:class:`fparser.one.block_statements.BeginSource`
358 
359  """

361  self._ast, self._name = KernelProcedure.get_procedure(
362  ktype_ast, ktype_name, modast)

364  @staticmethod
365  def get_procedure(ast, name, modast):
366  '''
367  Get the name of the subroutine associated with the Kernel. This is
368  a type-bound procedure in the meta-data which may take one of two
369  forms:
370  PROCEDURE, nopass :: code => <proc_name>
371  or
372  PROCEDURE, nopass :: <proc_name>
373 
374  :param ast: the fparser1 parse tree for the Kernel meta-data.
375  :type ast: :py:class:`fparser.one.block_statements.Type`
376  :param str name: the name of the Fortran type holding the Kernel \
377  meta-data.
378  :param modast: the fparser1 parse tree for the module containing the \
379  Kernel routine.
380  :type modast: :py:class:`fparser.one.block_statements.BeginSource`
381 
382  :returns: 2-tuple of the fparser1 parse tree of the Subroutine \
383  statement and the name of that Subroutine.
384  :rtype: (:py:class:`fparser1.block_statements.Subroutine`, str)
385 
386  :raises RuntimeError: if the supplied Kernel meta-data does not \
387  have a type-bound procedure.
388  :raises RuntimeError: if no implementation is found for the \
389  type-bound procedure.
390  :raises ParseError: if the type-bound procedure specifies a binding \
391  name but the generic name is not "code".
392  :raises ParseError: if the type-bound procedure is not public in the \
393  Fortran module.
394  :raises InternalError: if we get an empty string for the name of the \
395  type-bound procedure.
396  '''
397  bname = None
398  # Search the the meta-data for a SpecificBinding
399  for statement in ast.content:
400  if isinstance(statement, fparser1.statements.SpecificBinding):
401  # We support either:
402  # PROCEDURE, nopass :: code => <proc_name> or
403  # PROCEDURE, nopass :: <proc_name>
404  if statement.bname:
405  if statement.name.lower() != "code":
406  raise ParseError(
407  "Kernel type {0} binds to a specific procedure but"
408  " does not use 'code' as the generic name.".
409  format(name))
410  bname = statement.bname
411  else:
412  bname = statement.name
413  break
414  if bname is None:
415  raise RuntimeError(
416  "Kernel type {0} does not bind a specific procedure".
417  format(name))
418  if bname == '':
419  raise InternalError(
420  "Empty Kernel name returned for Kernel type {0}.".format(name))
421  code = None
422  default_public = True
423  declared_private = False
424  declared_public = False
425  for statement, depth in fpapi.walk(modast, -1):
426  if isinstance(statement, fparser1.statements.Private):
427  if len(statement.items) == 0:
428  default_public = False
429  elif bname in statement.items:
430  declared_private = True
431  if isinstance(statement, fparser1.statements.Public):
432  if len(statement.items) == 0:
433  default_public = True
434  elif bname in statement.items:
435  declared_public = True
436  if isinstance(statement, fparser1.block_statements.Subroutine) \
437  and statement.name == bname:
438  if statement.is_public():
439  declared_public = True
440  code = statement
441  if code is None:
442  raise RuntimeError("Kernel subroutine %s not implemented" % bname)
443  if declared_private or (not default_public and not declared_public):
444  raise ParseError("Kernel subroutine '%s' is not public" % bname)
445  return code, bname

477  def create(self, ast, name=None):
478  '''
479  Create a Kernel object for the API supplied to the constructor
480  of this factory.

483  :type ast: :py:class:`fparser.one.block_statements.BeginSource`
484  :param str name: the name of the Kernel or None.

493  elif self._type == "gocean1.0":
494  from psyclone.gocean1p0 import GOKernelType1p0
495  return GOKernelType1p0(ast, name=name)
496  else:
497  raise ParseError(
498  "KernelTypeFactory: Internal Error: Unsupported "
499  "kernel type '{0}' found. Should not be possible.".
500  format(self._type))

537 class KernelType(object):
538  """
539  Base class for describing Kernel Metadata.
540 
541  This contains the name of the elemental procedure and metadata associated
542  with how that procedure is mapped over mesh entities.
543 
544  :param ast: fparser1 AST for the parsed kernel meta-data.
545  :type ast: :py:class:`fparser.one.block_statements.BeginSource`
546  :param str name: name of the Fortran derived type describing the kernel.
547 
548  :raises ParseError: if the supplied name does not follow the convention \
549  of ending in "_mod" or the AST does not contain a \
550  module definition.
551  """
552  def __init__(self, ast, name=None):
553 
554  if name is None:
555  # if no name is supplied then use the module name to
556  # determine the type name. The assumed convention is that
557  # the module is called <name/>_mod and the type is called
558  # <name/>_type
559  found = False
560  for statement, depth in fpapi.walk(ast, -1):
561  if isinstance(statement, fparser1.block_statements.Module):
562  module_name = statement.name
563  found = True
564  break
565  if not found:
566  raise ParseError(
567  "Error KernelType, the file does not contain a module. "
568  "Is it a Kernel file?")
569 
570  mn_len = len(module_name)
571  if mn_len < 5:
572  raise ParseError(
573  "Error, module name '{0}' is too short to have '_mod' as "
574  "an extension. This convention is assumed.".
575  format(module_name))
576  base_name = module_name.lower()[:mn_len-4]
577  extension_name = module_name.lower()[mn_len-4:mn_len]
578  if extension_name != "_mod":
579  raise ParseError(
580  "Error, module name '{0}' does not have '_mod' as an "
581  "extension. This convention is assumed.".
582  format(module_name))
583  name = base_name + "_type"
584 
585  self._name = name
586  self._ast = ast
587  self.checkMetadataPublic(name, ast)
588  self._ktype = self.getKernelMetadata(name, ast)
589  self._iterates_over = self.get_integer_variable("iterates_over")
590  self._procedure = KernelProcedure(self._ktype, name, ast)
591  self._inits = self.getkerneldescriptors(self._ktype)
592  self._arg_descriptors = [] # this is set up by the subclasses
593 
594  def getkerneldescriptors(self, ast, var_name='meta_args'):
595  descs = ast.get_variable(var_name)
596  if descs is None:
597  raise ParseError(
598  "kernel call does not contain a {0} type".format(var_name))
599  try:
600  nargs = int(descs.shape[0])
601  except AttributeError:
602  raise ParseError(
603  "kernel metadata {0}: {1} variable must be an array".
604  format(self._name, var_name))
605  if len(descs.shape) is not 1:
606  raise ParseError(
607  "kernel metadata {0}: {1} variable must be a 1 dimensional "
608  "array".format(self._name, var_name))
609  if descs.init.find("[") is not -1 and descs.init.find("]") is not -1:
610  # there is a bug in f2py
611  raise ParseError(
612  "Parser does not currently support [...] initialisation for "
613  "{0}, please use (/.../) instead".format(var_name))
614  try:
615  inits = expr.FORT_EXPRESSION.parseString(descs.init)[0]
616  except ParseException:
617  raise ParseError("kernel metadata has an invalid format {0}".
618  format(descs.init))
619  nargs = int(descs.shape[0])
620  if len(inits) != nargs:
621  raise ParseError(
622  "Error, in {0} specification, the number of args {1} and "
623  "number of dimensions {2} do not match".
624  format(var_name, nargs, len(inits)))
625  return inits

671  def getKernelMetadata(self, name, ast):
