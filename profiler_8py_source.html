46 class Profiler(object):
47  ''' This class wraps all profiling related settings.'''
48 
49  # Command line option to use for the various profiling options
50  # INVOKES: Automatically add a region for each invoke. i.e. at
51  # the start and end of each PSyclone created subroutine.
52  # KERNELS: Automatically add a profile region around every
53  # kernel call including the loop structure created.
54  INVOKES = "invokes"
55  KERNELS = "kernels"
56  SUPPORTED_OPTIONS = [INVOKES, KERNELS]
57  _options = []
58  # A namespace manager to make sure we get unique region names
59  _namespace = NameSpace()
60 
61  # -------------------------------------------------------------------------
62  @staticmethod
63  def set_options(options):
64  '''Sets the option the user required.
65  :param options: List of options selected by the user, or None to
66  disable all automatic profiling.
67  :type options: List of strings.
68  :raises GenerationError: If any option is not KERNELS or INVOKES.
69  '''
70  # Test that all options are valid
71  if options is None:
72  options = [] # Makes it easier to test
73  for index, option in enumerate(options):
74  if option not in [Profiler.INVOKES, Profiler.KERNELS]:
75  # Create a 'nice' representation of the allowed options.
76  # [1:-1] cuts out the '[' and ']' that surrounding the
77  # string of the list.
78  allowed_options = str(Profiler.SUPPORTED_OPTIONS)[1:-1]
79  raise GenerationError("Error in Profiler.setOptions: options "
80  "must be one of {0} but found '{1}' "
81  "at {2}"
82  .format(allowed_options,
83  str(option), index))
84 
85  # Store options so they can be queried later
86  Profiler._options = options

105  @staticmethod
106  def add_profile_nodes(schedule, loop_class):
107  '''This function inserts all required Profiling Nodes (for invokes
108  and kernels, as specified on the command line) into a schedule.
109  :param schedule: The schedule to instrument.
110  :type schedule: :py::class::`psyclone.psyGen.Schedule` or derived class
111  :param loop_class: The loop class (e.g. GOLoop, DynLoop) to instrument.
112  :type loop_class: :py::class::`psyclone.psyGen.Loop` or derived class.
113  '''
114 
115  from psyclone.transformations import ProfileRegionTrans
116  profile_trans = ProfileRegionTrans()
117  if Profiler.profile_kernels():
118  for i in schedule.children:
119  if isinstance(i, loop_class):
120  profile_trans.apply(i)
121  if Profiler.profile_invokes():
122  profile_trans.apply(schedule.children)

140  def __init__(self, children=None, parent=None):
141  '''Constructor for a ProfileNode that is inserted in a schedule.
142  Parameters:
143  :param children: A list of children nodes for this node.
144  :type children: A list of :py::class::`psyclone.psyGen.Node` \
145  or derived classes.
146  :param parent: The parent of this node.
147  :type parent: A :py::class::`psyclone.psyGen.Node`.
148  '''
149  Node.__init__(self, children=children, parent=parent)
150 
151  # Store the name of the profile variable that is used for this
152  # profile name. This allows to show the variable name in __str__
153  # (and also if we would call create_name in gen(), the name would
154  # change every time gen() is called).
155  self._var_name = NameSpaceFactory().create().create_name("profile")
156 
157  # Name of the region. In general at constructor time we might not
158  # have a parent subroutine or a child for the kernel, so we leave
159  # the name empty for now. The region and module names are set the
160  # first time gen() is called (and then remain unchanged).
161  self._region_name = None
162  self._module_name = None
163 
164  # -------------------------------------------------------------------------
165  def __str__(self):
166  ''' Returns a string representation of the subtree starting at
167  this node. '''
168  result = "ProfileStart[var={0}]\n".format(self._var_name)
169  for child in self.children:
170  result += str(child)+"\n"
171  return result+"ProfileEnd"

196  def gen_code(self, parent):
197  # pylint: disable=arguments-differ
198  '''Creates the profile start and end calls, surrounding the children
199  of this node.
200  :param parent: The parent of this node.
201  :type parent: :py:class:`psyclone.psyGen.Node`.'''
202 
203  if self._module_name is None or self._region_name is None:
204  # Find the first kernel and use its name. In an untransformed
205  # Schedule there should be only one kernel, but if Profile is
206  # invoked after e.g. a loop merge more kernels might be there.
207  region_name = "unknown-kernel"
208  module_name = "unknown-module"
209  for kernel in self.walk(self.children, Kern):
210  region_name = kernel.name
211  module_name = kernel.module_name
212  break
213  if self._region_name is None:
214  self._region_name = Profiler.create_unique_region(region_name)
215  if self._module_name is None:
216  self._module_name = module_name
217 
218  # Note that adding a use statement makes sure it is only
219  # added once, so we don't need to test this here!
220  use = UseGen(parent, "profile_mod", only=True,
221  funcnames=["ProfileData, ProfileStart, ProfileEnd"])
222  parent.add(use)
223  prof_var_decl = TypeDeclGen(parent, datatype="ProfileData",
224  entity_decls=[self._var_name],
225  save=True)
226  parent.add(prof_var_decl)
227 
228  prof_start = CallGen(parent, "ProfileStart",
229  ["\"{0}\"".format(self._module_name),
230  "\"{0}\"".format(self._region_name),
231  self._var_name])
232  parent.add(prof_start)
233 
234  for child in self.children:
235  child.gen_code(parent)
236 
237  prof_end = CallGen(parent, "ProfileEnd",
238  [self._var_name])
239  parent.add(prof_end)
