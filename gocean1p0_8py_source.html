144 class GOInvokes(Invokes):
145  '''
146  The GOcean specific invokes class. This passes the GOcean specific
147  invoke class to the base class so it creates the one we require.
148  :param alg_calls: The Invoke calls discovered in the Algorithm layer.
149  :type alg_calls: OrderedDict of :py:class:`psyclone.parse.InvokeCall` \
150  objects.
151  '''
152  def __init__(self, alg_calls):
153  if False: # pylint: disable=using-constant-test
154  self._0_to_n = GOInvoke(None, None) # for pyreverse
155  Invokes.__init__(self, alg_calls, GOInvoke)
156 
157  index_offsets = []
158  # Loop over all of the kernels in all of the invoke() calls
159  # and check that they work on compatible grid-index offsets.
160  # Strictly speaking this check should be done in the parsing
161  # code since it is a check on the correctness of the meta-data.
162  # However, that would require a fundamental change to the parsing
163  # code since it requires information on all of the invokes and
164  # kernels in an application. Therefore it is much simpler to
165  # do it here where we have easy access to that information.
166  for invoke in self.invoke_list:
167  for kern_call in invoke.schedule.kern_calls():
168  # We only care if the index offset is not offset_any (since
169  # that is compatible with any other offset)
170  if kern_call.index_offset != "go_offset_any":
171  # Loop over the offsets we've seen so far
172  for offset in index_offsets:
173  if offset != kern_call.index_offset:
174  raise GenerationError(
175  "Meta-data error in kernel {0}: "
176  "INDEX_OFFSET of '{1}' does not match that "
177  "({2}) of other kernels. This is not "
178  "supported.".format(kern_call.name,
179  kern_call.index_offset,
180  offset))
181  # Append the index-offset of this kernel to the list of
182  # those seen so far
183  index_offsets.append(kern_call.index_offset)

252  def gen_code(self, parent):
253  '''
254  Generates GOcean specific invocation code (the subroutine called
255  by the associated invoke call in the algorithm layer). This
256  consists of the PSy invocation subroutine and the declaration of
257  its arguments.
258 
259  :param parent: the node in the generated AST to which to add content.
260  :type parent: :py:class:`psyclone.f2pygen.ModuleGen`
261  '''
262  from psyclone.f2pygen import SubroutineGen, DeclGen, TypeDeclGen, \
263  CommentGen, AssignGen
264  # create the subroutine
265  invoke_sub = SubroutineGen(parent, name=self.name,
266  args=self.psy_unique_var_names)
267  parent.add(invoke_sub)
268 
269  # add declarations for the variables holding the upper bounds
270  # of loops in i and j
271  if self.schedule.const_loop_bounds:
272  invoke_sub.add(DeclGen(invoke_sub, datatype="INTEGER",
273  entity_decls=[self.schedule.iloop_stop,
274  self.schedule.jloop_stop]))
275 
276  # Generate the code body of this subroutine
277  self.schedule.gen_code(invoke_sub)
278 
279  # If we're generating an OpenCL routine then the arguments must
280  # have the target attribute as we pass pointers to them in to
281  # the OpenCL run-time.
282  target = bool(self.schedule.opencl)
283 
284  # add the subroutine argument declarations for fields
285  if self.unique_args_arrays:
286  my_decl_arrays = TypeDeclGen(invoke_sub, datatype="r2d_field",
287  intent="inout", target=target,
288  entity_decls=self.unique_args_arrays)
289  invoke_sub.add(my_decl_arrays)
290 
291  # add the subroutine argument declarations for real scalars
292  if self.unique_args_rscalars:
293  my_decl_rscalars = DeclGen(invoke_sub, datatype="REAL",
294  intent="inout", kind="go_wp",
295  entity_decls=self.unique_args_rscalars)
296  invoke_sub.add(my_decl_rscalars)
297  # add the subroutine argument declarations for integer scalars
298  if self.unique_args_iscalars:
299  my_decl_iscalars = DeclGen(invoke_sub, datatype="INTEGER",
300  intent="inout",
301  entity_decls=self.unique_args_iscalars)
302  invoke_sub.add(my_decl_iscalars)
303 
304  if self._schedule.const_loop_bounds and self.unique_args_arrays:
305 
306  # Look-up the loop bounds using the first field object in the
307  # list
308  sim_domain = self.unique_args_arrays[0] +\
309  "%grid%subdomain%internal%"
310  position = invoke_sub.last_declaration()
311 
312  invoke_sub.add(CommentGen(invoke_sub, ""),
313  position=["after", position])
314  invoke_sub.add(AssignGen(invoke_sub, lhs=self.schedule.jloop_stop,
315  rhs=sim_domain+"ystop"),
316  position=["after", position])
317  invoke_sub.add(AssignGen(invoke_sub, lhs=self.schedule.iloop_stop,
318  rhs=sim_domain+"xstop"),
319  position=["after", position])
320  invoke_sub.add(CommentGen(invoke_sub, " Look-up loop bounds"),
321  position=["after", position])
322  invoke_sub.add(CommentGen(invoke_sub, ""),
323  position=["after", position])

523  @staticmethod
524  def add_bounds(bound_info):
525  '''
526  Adds a new iteration space to PSyclone. An iteration space in the
527  gocean1.0 API is for a certain offset type and field type. It defines
528  the loop boundaries for the outer and inner loop. The format is a
529  ":" separated tuple:
530  bound_info = offset-type:field-type:iteration-space:outer-start:
531  outer-stop:inner-start:inner-stop
532  Example:
533  bound_info = go_offset_ne:go_ct:go_all_pts\
534  :{start}-1:{stop}+1:{start}:{stop}
535 
536  The expressions {start} and {stop} will be replaced with the loop
537  indices that correspond to the inner points (i.e. non-halo or
538  boundary points) of the field. So the index {start}-1 is actually
539  on the halo / boundary.
540 
541  :param str bound_info: A string that contains a ":" separated \
542  tuple with the iteration space definition.
543  :raises ValueError: if bound_info is not a string.
544  :raises ConfigurationError: if bound_info is not formatted correctly.
545  '''
546 
547  if not isinstance(bound_info, str):
548  raise InternalError("The parameter 'bound_info' must be a string, "
549  "got '{0}' (type {1})"
550  .format(bound_info, type(bound_info)))
551 
552  data = bound_info.split(":")
553  if len(data) != 7:
554  from psyclone.configuration import ConfigurationError
555  raise ConfigurationError("An iteration space must be in the form "
556  "\"offset-type:field-type:"
557  "iteration-space:outer-start:"
558  "outer-stop:inner-start:inner-stop\"\n"
559  "But got \"{0}\"".format(bound_info))

1469  j

1472  |
