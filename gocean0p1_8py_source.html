46 class GOPSy(PSy):
47  '''
48  The GOcean specific PSy class. This creates a GOcean specific
49  invokes object (which controls all the required invocation calls).
50  Also overrides the PSy gen method so that we generate GOceaen
51  specific PSy module code.
52 
53  :param invoke_info: An object containing the required invocation \
54  information for code optimisation and generation.
55  :type invoke_info: :py:class:`psyclone.parse.FileInfo`
56  '''
57  def __init__(self, invoke_info):
58  PSy.__init__(self, invoke_info)
59  self._invokes = GOInvokes(invoke_info.calls)

61  @property
62  def gen(self):
63  '''
64  Generate PSy code for the GOcean api.
65 
66  :rtype: ast
67 
68  '''
69  from psyclone.f2pygen import ModuleGen, UseGen
70 
71  # create an empty PSy layer module
72  psy_module = ModuleGen(self.name)
73  # include the kind_params module
74  psy_module.add(UseGen(psy_module, name="kind_params_mod"))
75  # include the topology_mod module for loop bounds
76  psy_module.add(UseGen(psy_module, name="topology_mod"))
77  # include the field_mod module in case we have any r-space variables
78  psy_module.add(UseGen(psy_module, name="field_mod",
79  only=["scalar_field_type"]))
80  # add in the subroutines for each invocation
81  self.invokes.gen_code(psy_module)
82  # inline kernels where requested
83  self.inline(psy_module)
84  return psy_module.root

112  @property
113  def unique_args_arrays(self):
114  ''' find unique arguments that are arrays (defined as those that are
115  not rspace). GOcean needs to kow this as we are dealing with
116  arrays directly so need to declare them correctly. '''
117  result = []
118  for call in self._schedule.calls():
119  for arg in call.arguments.args:
120  if(not arg.is_literal and
121  not arg.space.lower() == "r" and
122  arg.name not in result):
123  result.append(arg.name)
124  return result

234 class GOKernCallFactory(object):
235  ''' A GOcean-specific kernel call factory. '''
236  @staticmethod
237  def create(call, parent=None):
238  ''' Creates a kernel call and associated Loop structure '''
239  outer_loop = GOLoop(parent=parent,
240  loop_type="outer")
241  inner_loop = GOLoop(parent=outer_loop,
242  loop_type="inner")
243  outer_loop.addchild(inner_loop)
244  gocall = GOKern()
245  gocall.load(call, parent=inner_loop)
246  inner_loop.addchild(gocall)
247  # determine inner and outer loops space information from the
248  # child kernel call. This is only picked up automatically (by
249  # the inner loop) if the kernel call is passed into the inner
250  # loop.
251  inner_loop.iteration_space = gocall.iterates_over
252  outer_loop.iteration_space = inner_loop.iteration_space
253  inner_loop.field_space = gocall.arguments.\
254  iteration_space_arg().function_space
255  outer_loop.field_space = inner_loop.field_space
256  inner_loop.field_name = gocall.arguments.\
257  iteration_space_arg().name
258  outer_loop.field_name = inner_loop.field_name
259  return outer_loop
