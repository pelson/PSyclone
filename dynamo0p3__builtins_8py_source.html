74 class DynBuiltInCallFactory(object):
75  ''' Creates the necessary framework for a call to a Dynamo built-in,
76  This consists of the operation itself and the loop over unique DoFs. '''
77 
78  def __str__(self):
79  return "Factory for a call to a Dynamo built-in"
80 
81  @staticmethod
82  def create(call, parent=None):
83  ''' Create the objects needed for a call to the built-in
84  described in the call (BuiltInCall) object '''
85 
86  if call.func_name not in BUILTIN_MAP:
87  raise ParseError(
88  "Unrecognised built-in call. Found '{0}' but expected "
89  "one of '{1}'".format(call.func_name,
90  list(BUILTIN_MAP_CAPITALISED.keys())))
91 
92  # Use our dictionary to get the correct Python object for
93  # this built-in.
94  builtin = BUILTIN_MAP[call.func_name]()
95 
96  # Create the loop over DoFs
97  dofloop = DynLoop(parent=parent,
98  loop_type="dofs")
99 
100  # Use the call object (created by the parser) to set-up the state
101  # of the infrastructure kernel
102  builtin.load(call, parent=dofloop)
103 
104  # Check that our assumption that we're looping over DoFS is valid
105  if builtin.iterates_over != "dofs":
106  raise NotImplementedError(
107  "In the Dynamo 0.3 API built-in calls must iterate over "
108  "DoFs but found {0} for {1}".format(builtin.iterates_over,
109  str(builtin)))
110  # Set-up its state
111  dofloop.load(builtin)
112  # As it is the innermost loop it has the kernel as a child
113  dofloop.addchild(builtin)
114 
115  # Return the outermost loop
116  return dofloop

119 class DynBuiltIn(BuiltIn):
120  ''' Parent class for a call to a Dynamo Built-in. '''
121 
122  def __str__(self):
123  raise NotImplementedError("DynBuiltIn.__str__ must be overridden")
124 
125  def __init__(self):
126  self._name_space_manager = NameSpaceFactory().create()
127  # Look=up/create the name of the loop variable for the loop over DoFs
128  self._idx_name = self._name_space_manager.\
129  create_name(root_name="df",
130  context="PSyVars",
131  label="dof_loop_idx")
132  BuiltIn.__init__(self)
133 
134  def load(self, call, parent=None):
135  ''' Populate the state of this object using the supplied call
136  object. '''
137  from psyclone.dynamo0p3 import FSDescriptors
138  BuiltIn.load(self, call, DynKernelArguments(call, self), parent)
139  self.arg_descriptors = call.ktype.arg_descriptors
140  self._func_descriptors = call.ktype.func_descriptors
141  self._fs_descriptors = FSDescriptors(call.ktype.func_descriptors)
142  # Check that this built-in kernel is valid
143  self._validate()
144 
145  def _validate(self):
146  ''' Check that this built-in conforms to the Dynamo 0.3 API '''
147  write_count = 0 # Only one argument must be written to
148  field_count = 0 # We must have one or more fields as arguments
149  spaces = set() # All field arguments must be on the same space
150  for arg in self.arg_descriptors:
151  if arg.access in ["gh_write", "gh_sum", "gh_inc"]:
152  write_count += 1
153  if arg.type == "gh_field":
154  field_count += 1
155  spaces.add(arg.function_space)
156  if arg.type not in VALID_BUILTIN_ARG_TYPES:
157  raise ParseError(
158  "In the Dynamo 0.3 API an argument to a built-in kernel "
159  "must be one of {0} but kernel {1} has an argument of "
160  "type {2}".format(VALID_BUILTIN_ARG_TYPES, self.name,
161  arg.type))
162  if write_count != 1:
163  raise ParseError("A built-in kernel in the Dynamo 0.3 API must "
164  "have one and only one argument that is written "
165  "to but found {0} for kernel {1}".
166  format(write_count, self.name))
167  if field_count == 0:
168  raise ParseError("A built-in kernel in the Dynamo 0.3 API "
169  "must have at least one field as an argument but "
170  "kernel {0} has none.".format(self.name))
171  if len(spaces) != 1:
172  raise ParseError(
173  "All field arguments to a built-in in the Dynamo 0.3 API "
174  "must be on the same space. However, found spaces {0} for "
175  "arguments to {1}".format(sorted(spaces), self.name))
