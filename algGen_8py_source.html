29 class Alg(object):
30  '''Generate a modified algorithm code for a single algorithm
31  specification. Takes the ast of the algorithm specification output
32  from the function :func:`parse.parse` and an instance of the
33  :class:`psyGen.PSy` class as input.
34 
35  :param ast ast: An object containing an ast of the algorithm
36  specification which was produced by the function
37  :func:`parse.parse`.
38 
39  :param PSy psy: An object (:class:`psyGen.PSy`) containing
40  information about the PSy layer.
41 
42  For example:
43 
44  >>> from parse import parse
45  >>> ast,info=parse("argspec.F90")
46  >>> from psyGen import PSy
47  >>> psy=PSy(info)
48  >>> from algGen import Alg
49  >>> alg=Alg(ast,psy)
50  >>> print(alg.gen)
51 
52  '''
53 
54  def __init__(self, ast, psy):
55  self._ast = ast
56  self._psy = psy
57 
58  @property
59  def gen(self):
60  '''
61  Generate modified algorithm code
62 
63  :rtype: ast
64 
65  '''
66  from fparser import api
67  from psyclone.f2pygen import adduse
68  psy_name = self._psy.name
69  # run through all statements looking for procedure calls
70  idx = 0
71  for stmt, _ in api.walk(self._ast, -1):
72 
73  if isinstance(stmt, fparser.one.statements.Call):
74  if stmt.designator == "invoke":
75  invoke_info = self._psy.invokes.invoke_list[idx]
76  stmt.designator = invoke_info.name
77  stmt.items = invoke_info.alg_unique_args
78  adduse(psy_name, stmt.parent, only=True,
79  funcnames=[invoke_info.name])
80  idx += 1
81 
82  if idx == 0:
83  raise NoInvokesError(
84  "Algorithm file contains no invoke() calls: refusing to "
85  "generate empty PSy code")
86 
87  return self._ast
