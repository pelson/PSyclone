55 def generate(filename, api=""):
56 
57  '''Generates an empty kernel subroutine with the required arguments
58  and datatypes (which we call a stub) when presented with Kernel
59  Metadata. This is useful for Kernel developers to make sure
60  they are using the correct arguments in the correct order. The
61  Kernel Metadata must be presented in the standard Kernel
62  format.
63 
64  :param str filename: The name of the file for which to create a \
65  kernel stub for.
66  :param str api: The name of the API for which to create a kernel \
67  stub. Must be one of the supported stub APIs.
68 
69  :raise GenerationError: if an invalid stub API is specified.
70  :raise IOError: if filename does not specify a file.
71  :raise ParseError: if the given file could not be parsed.
72  '''
73  if api == "":
74  api = Config.get().default_stub_api
75  if api not in Config.get().supported_stub_apis:
76  print("Unsupported API '{0}' specified. Supported API's are {1}.".
77  format(api, Config.get().supported_stub_apis))
78  raise GenerationError(
79  "generate: Unsupported API '{0}' specified. Supported types are "
80  "{1}.".format(api, Config.get().supported_stub_apis))
81 
82  if not os.path.isfile(filename):
83  raise IOError("file '{0}' not found".format(filename))
84 
85  # drop cache
86  fparser.one.parsefortran.FortranParser.cache.clear()
87  fparser.logging.disable(fparser.logging.CRITICAL)
88  try:
89  ast = fparser.api.parse(filename, ignore_comments=False)
90 
91  except (fparser.common.utils.AnalyzeError, AttributeError) as error:
92  raise ParseError("Code appears to be invalid Fortran: " +
93  str(error))
94 
95  metadata = DynKernMetadata(ast)
96  kernel = DynKern()
97  kernel.load_meta(metadata)
98  return kernel.gen_stub

101 def run():
102  ''' Top-level driver for the kernel-stub generator. Handles command-line
103  flags, calls generate() and applies line-length limiting to the output (if
104  requested). '''
105  import argparse
106  parser = argparse.ArgumentParser(description="Create Kernel stub code from"
107  " Kernel metadata")
108  parser.add_argument("-o", "--outfile", help="filename of output")
109  parser.add_argument("-api", default=Config.get().default_stub_api,
110  help="choose a particular api from {0}, default {1}".
111  format(str(Config.get().supported_stub_apis),
112  Config.get().default_stub_api))
113  parser.add_argument('filename', help='Kernel metadata')
114  parser.add_argument(
115  '-l', '--limit', dest='limit', action='store_true', default=False,
116  help='limit the fortran line length to 132 characters')
117 
118  args = parser.parse_args()
119 
120  try:
121  stub = generate(args.filename, api=args.api)
122  except (IOError, ParseError, GenerationError, RuntimeError) as error:
123  print("Error:", error)
124  exit(1)
125  except Exception as error: # pylint: disable=broad-except
126  print("Error, unexpected exception:\n")
127  exc_type, exc_value, exc_traceback = sys.exc_info()
128  print(exc_type)
129  print(exc_value)
130  traceback.print_tb(exc_traceback)
131  exit(1)
132 
133  if args.limit:
134  fll = FortLineLength()
135  stub_str = fll.process(str(stub))
136  else:
137  stub_str = str(stub)
138  if args.outfile is not None:
139  my_file = open(args.outfile, "w")
140  my_file.write(stub_str)
141  my_file.close()
142  else:
143  print("Kernel stub code:\n", stub_str)
