18 class ExpressionNode(object):
19  '''Base class for all expression tree nodes'''
20  def __init__(self, toks):
21  ''' The recursive collection of names enables the dependencies of
22  expressions to be analysed. '''
23  self.names = set()
24  for tok in toks:
25  if isinstance(tok, ExpressionNode):
26  self.names.update(tok.names)
27 
28  # Keep the list of toks for future reference.
29  self.toks = toks
30 
31  def walk_skipping_name(self):
32  ''' Generator for depth-first walk of this expression, skipping
33  the name part of the walk. '''
34  for tok in self.toks:
35  if isinstance(tok, ExpressionNode):
36  for index, i in enumerate(tok.walk_skipping_name()):
37  if index > 0:
38  yield i
39  yield self

56 class BinaryOperator(ExpressionNode):
57  '''Expression node for one or more binary operators with the same
58  precedence.
59 
60  For some reason, operator tokens come in a list of lists.'''
61  def __init__(self, toks):
62  ExpressionNode.__init__(self, toks[0])
63 
64  self.operands = [toks[0][0]]
65  self.symbols = []
66  i = iter(toks[0][1:])
67  try:
68  while True:
69  tok = next(i)
70  self.symbols.append(tok)
71  tok = next(i)
72  self.operands.append(tok)
73  except StopIteration:
74  pass
75 
76  def __repr__(self):
77  _str = "BinaryOperator([["+repr(self.operands[0])
78  for sym, opd in zip(self.symbols, self.operands[1:]):
79  _str += ", " + repr(sym) + ", " + repr(opd)
80  _str += "]])"
81  return _str
82 
83  def __str__(self):
84  return str(self.operands[0]) + \
85  "".join([" "+str(sym)+" "+str(opd) for sym, opd in
86  zip(self.symbols, self.operands[1:])])

199 class NamedArg(ExpressionNode):
200  ''' Expression node for a Fortran named argument. '''
201  def __init__(self, toks):
202  ExpressionNode.__init__(self, toks)
203 
204  # First token is the name of the argument
205  self._name = toks[0]
206  self.names.update([self._name])
207 
208  # The second token is the '=' so we ignore that and skip to
209  # the third token which contains the value assigned to the
210  # argument...
211  # The named variable can be assigned a character string. We've
212  # told the parser not to remove the delimiters so that we can
213  # see whether they are single or double quotes.
214  first_char = toks[2][0]
215  if first_char in ["'", '"']:
216  # Store the quotation character and the content of the string
217  # excluding the quotation marks
218  self._quote = toks[2][0]
219  self._value = toks[2][1:-1]
220  else:
221  # The quantity being assigned is not a string
222  self._quote = None
223  self._value = toks[2]
224 
225  def __repr__(self):
226  if self._quote:
227  if self._quote == "'":
228  _str = "NamedArg(['{0}', '=', \"'{1}'\"])".format(self._name,
229  self._value)
230  else:
231  _str = 'NamedArg(["{0}", "=", \'"{1}"\'])'.format(self._name,
232  self._value)
233  else:
234  _str = "NamedArg(['{0}', '=', '{1}'])".format(self._name,
235  self._value)
236  return _str
237 
238  def __str__(self):
239  _str = str(self._name) + "="
240  if self._quote:
241  _str += self._quote + str(self._value) + self._quote
242  else:
243  _str += str(self._value)
244  return _str

263 # Construct a grammar using PyParsing
264 
265 # A Fortran variable name starts with a letter and continues with
266 # letters, numbers and _. Can you start a name with _?
267 VAR_NAME = pparse.Word(pparse.alphas, pparse.alphanums+"_")
268 NAME = VAR_NAME | pparse.Literal(".false.") | pparse.Literal(".true.")
269 
270 # Reference to a component of a derived type
271 DERIVED_TYPE_COMPONENT = pparse.Combine(VAR_NAME + "%" + VAR_NAME)
272 
273 # An unsigned integer
274 UNSIGNED = pparse.Word(pparse.nums)
275 
276 # In Fortran, a numerical constant can have its kind appended after an
277 # underscore. The kind can be a 'name' or just digits.
278 KIND = pparse.Word("_", exact=1) + (VAR_NAME | UNSIGNED)
279 
280 # First arg to Word gives allowed initial chars, 2nd arg gives allowed
281 # body characters
282 SIGNED = pparse.Word("+-"+pparse.nums, pparse.nums)
283 INTEGER = pparse.Combine(SIGNED + pparse.Optional(KIND))
284 
285 POINT = pparse.Literal(".")
286 
287 # A floating point number
288 REAL = pparse.Combine(
289  (SIGNED + POINT + pparse.Optional(UNSIGNED) | POINT + UNSIGNED) +
290  pparse.Optional(pparse.Word("dDeE", exact=1) + SIGNED) +
291  pparse.Optional(KIND))
292 
293 # Literal brackets.
294 LPAR = pparse.Literal("(")
295 RPAR = pparse.Literal(")")
296 
297 LIT_ARRAY_START = pparse.Literal("[") | pparse.Literal("(/")
298 LIT_ARRAY_END = pparse.Literal("]") | pparse.Literal("/)")
299 
300 EXPR = pparse.Forward()
301 
302 # Array slicing
303 COLON = pparse.Literal(":")
304 SLICING = pparse.Optional(EXPR) + COLON + pparse.Optional(EXPR) + \
305  pparse.Optional(COLON+pparse.Optional(EXPR))
306 SLICING.setParseAction(lambda strg, loc, toks: [Slicing(toks)])
307 
308 VAR_OR_FUNCTION = (DERIVED_TYPE_COMPONENT | NAME) + pparse.Optional(
309  LPAR + pparse.Optional(pparse.delimitedList(SLICING | EXPR)) + RPAR)
310 VAR_OR_FUNCTION.setParseAction(lambda strg, loc, toks: [FunctionVar(toks)])
311 
312 LITERAL_ARRAY = LIT_ARRAY_START + pparse.delimitedList(EXPR) + LIT_ARRAY_END
313 LITERAL_ARRAY.setParseAction(lambda strg, loc, toks: [LiteralArray(toks)])
314 
315 # An optional/named argument. We use QuotedString here to avoid versioning
316 # problems with the interface to {sgl,dbl}QuotedString in pyparsing.
317 OPTIONAL_VAR = VAR_NAME + "=" + ((NAME | REAL | INTEGER) |
318  pparse.QuotedString("'",
319  unquoteResults=False) |
320  pparse.QuotedString('"',
321  unquoteResults=False))
322 # lambda creates a temporary function which, in this case, takes three
323 # arguments and creates a NamedArg object.
324 OPTIONAL_VAR.setParseAction(lambda strg, loc, toks: [NamedArg(toks)])
325 
326 GROUP = LPAR + EXPR + RPAR
327 GROUP.setParseAction(lambda strg, loc, toks: [Grouping(toks)])
328 
329 # Parser will attempt to match with the expressions in the order they
330 # are specified here. Therefore must list them in order of decreasing
331 # generality
332 OPERAND = (GROUP | OPTIONAL_VAR | VAR_OR_FUNCTION | REAL | INTEGER |
333  LITERAL_ARRAY)
334 
335 # Cause the binary operators to work.
336 OPERATOR = pparse.operatorPrecedence(
337  OPERAND,
338  ((pparse.Literal("**"), 2, pparse.opAssoc.RIGHT,
339  lambda strg, loc, toks: [BinaryOperator(toks)]),
340  (pparse.Literal("*") | pparse.Literal("/"), 2, pparse.opAssoc.LEFT,
341  lambda strg, loc, toks: [BinaryOperator(toks)]),
342  (pparse.Literal("+") | pparse.Literal("-"), 2, pparse.opAssoc.LEFT,
343  lambda strg, loc, toks: [BinaryOperator(toks)]),))
344 
345 EXPR << (OPERATOR | OPERAND)
